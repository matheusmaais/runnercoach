# Running Coach V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build V1 of the personal running coach system: local data contracts, Garmin ingestion, check-in YAML, adaptive plan/recommendation guardrails, science registry, reports, and a modern local dashboard.

**Architecture:** Implement a local Python package with typed models and deterministic transforms. Raw Garmin data stays local/ignored; canonical manual inputs live as YAML; processed CSVs, docs, dashboard, and decisions are generated by a single pipeline command. The recommendation engine adapts planned workouts inside numbered weeks and deterministic guardrails before any LLM layer is added.

**Tech Stack:** Python 3.11+, `pytest`, `pydantic`, `PyYAML`, standard `csv/json/hashlib/zoneinfo`, `openpyxl` for V1 dashboard generation, Markdown docs, Makefile wrappers.

---

## Source Spec

Primary design spec:

- `docs/superpowers/specs/2026-05-29-running-coach-system-design.md`

Implementation must preserve these invariants:

- Garmin HR/cadence/power/running dynamics always belong to Matheus.
- Pace/distance/time are shared only when `shared_run=true` and `bruna_present=true`.
- Matheus solo activity cannot update Bruna zones, projections, or readiness.
- Red-flag symptoms trigger conservative recovery and suppress performance escalation.
- Recommendations must reference plan context: phase, week number, planned workout, and fallback.
- Science citations come only from `data/knowledge/science_refs.yaml`.
- `data/processed/science_refs.csv` is derived and must never overwrite the canonical YAML.
- CSV multi-value fields are compact JSON strings.
- Timezone is `America/Sao_Paulo`; weeks start Monday.

```python
from __future__ import annotations

import json
from typing import Any


def encode_json_cell(value: list[str] | list[dict[str, Any]]) -> str:
    return json.dumps(value, ensure_ascii=False, separators=(",", ":"))


def decode_json_cell(value: str) -> list[Any]:
    if value == "" or value is None:
        return []
    decoded = json.loads(value)
    if not isinstance(decoded, list):
        raise ValueError("CSV JSON cell must decode to a list")
    return decoded
```

```python
from __future__ import annotations

import csv
import hashlib
import io
import re
from datetime import datetime
from typing import Any

from running_coach.models import TIMEZONE


def parse_duration_seconds(value: str) -> float:
    hours, minutes, seconds = value.split(":")
    return int(hours) * 3600 + int(minutes) * 60 + float(seconds)


def parse_number(value: str) -> float | None:
    if value in {"", "--", None}:
        return None
    return float(str(value).replace(",", "."))


def parse_int(value: str) -> int | None:
    number = parse_number(value)
    return int(number) if number is not None else None


def make_activity_id(local_datetime: str, distance_km: float, duration_seconds: float, title: str) -> str:
    dt = datetime.strptime(local_datetime, "%Y-%m-%d %H:%M:%S")
    normalized_title = re.sub(r"\\s+", " ", title.strip().lower())
    digest = hashlib.sha1(normalized_title.encode("utf-8")).hexdigest()[:8]
    distance_token = f"{distance_km:.2f}".replace(".", "p")
    duration_token = str(int(round(duration_seconds)))
    return f"garmin-{dt:%Y%m%dT%H%M%S}-{distance_token}km-{duration_token}s-{digest}"


def parse_garmin_csv_text(csv_text: str, source_file: str) -> list[dict[str, Any]]:
    reader = csv.DictReader(io.StringIO(csv_text))
    output: list[dict[str, Any]] = []
    for index, row in enumerate(reader, start=2):
        local_datetime = row["Data"]
        local_date = local_datetime.split(" ")[0]
        distance_km = parse_number(row.get("Distância", "")) or 0.0
        duration_seconds = parse_duration_seconds(row.get("Tempo", "00:00:00"))
        title = row.get("Título", "")
        output.append(
            {
                "activity_id": make_activity_id(local_datetime, distance_km, duration_seconds, title),
                "source_file": source_file,
                "source_row_number": index,
                "local_date": local_date,
                "local_datetime": local_datetime,
                "timezone": TIMEZONE,
                "activity_type": row.get("Tipo de atividade", ""),
                "title": title,
                "distance_km": distance_km,
                "duration_seconds": duration_seconds,
                "matheus_avg_hr": parse_int(row.get("FC Média", "")),
                "matheus_max_hr": parse_int(row.get("FC máxima", "")),
                "avg_pace": row.get("Ritmo médio", ""),
                "best_pace": row.get("Melhor ritmo", ""),
                "matheus_cadence": parse_int(row.get("Cadência de corrida média", "")),
                "matheus_power": parse_int(row.get("Potência média", "")),
                "matheus_ground_contact": parse_number(row.get("Tempo médio de contato com o solo", "")),
                "matheus_stride_length": parse_number(row.get("Comprimento médio da passada", "")),
                "data_owner_hr": "matheus",
                "data_owner_dynamics": "matheus",
                "is_shared_run_candidate": row.get("Tipo de atividade", "") == "Corrida",
            }
        )
    return output
```

```python
from __future__ import annotations

from pydantic import BaseModel

from running_coach.models import Confidence, RecommendationAction, SymptomSeverity


class RecommendationInput(BaseModel):
    bruna_pse: int | None
    symptom_severity: SymptomSeverity
    matheus_achilles_morning: int
    matheus_achilles_after: int
    volleyball_previous_day: bool
    poor_sleep: bool
    all_out_race: bool
    planned_action: RecommendationAction
    phase: str
    week_number: int
    planned_workout_id: str


class RecommendationResult(BaseModel):
    action: RecommendationAction
    confidence: Confidence
    blocked_by_red_flag: bool
    reasons: list[str]
    missing_evidence: list[str]
    assumptions: list[str]
    phase: str
    week_number: int
    planned_workout_id: str


def recommend_next_action(input_data: RecommendationInput) -> RecommendationResult:
    reasons: list[str] = []
    missing: list[str] = []
    assumptions: list[str] = []

    if input_data.bruna_pse is None:
        missing.append("bruna_pse")
        assumptions.append("Bruna intensity inferred from shared pace and context.")

    if input_data.symptom_severity == SymptomSeverity.RED_FLAG:
        reasons.append("red_flag_symptom")
        return RecommendationResult(
            action=RecommendationAction.REPLACE_WITH_OFF,
            confidence=Confidence.HIGH,
            blocked_by_red_flag=True,
            reasons=reasons,
            missing_evidence=missing,
            assumptions=assumptions,
            phase=input_data.phase,
            week_number=input_data.week_number,
            planned_workout_id=input_data.planned_workout_id,
        )

    if input_data.matheus_achilles_morning >= 5 or input_data.matheus_achilles_after >= 5:
        reasons.append("matheus_achilles_ge_5")
        return RecommendationResult(
            action=RecommendationAction.BRUNA_WITHOUT_MATHEUS,
            confidence=Confidence.HIGH,
            blocked_by_red_flag=False,
            reasons=reasons,
            missing_evidence=missing,
            assumptions=assumptions,
            phase=input_data.phase,
            week_number=input_data.week_number,
            planned_workout_id=input_data.planned_workout_id,
        )

    if input_data.bruna_pse is not None and input_data.bruna_pse >= 9:
        reasons.append("bruna_pse_ge_9")
        action = RecommendationAction.REPLACE_WITH_EASY
    elif input_data.matheus_achilles_morning >= 3 or input_data.matheus_achilles_after >= 3:
        reasons.append("matheus_achilles_ge_3")
        action = RecommendationAction.DEFER_QUALITY
    elif input_data.volleyball_previous_day or input_data.poor_sleep or input_data.all_out_race:
        reasons.append("load_or_recovery_modifier")
        action = RecommendationAction.REDUCE_NEXT_WORKOUT
    else:
        reasons.append("within_guardrails")
        action = input_data.planned_action

    return RecommendationResult(
        action=action,
        confidence=Confidence.MEDIUM if missing else Confidence.HIGH,
        blocked_by_red_flag=False,
        reasons=reasons,
        missing_evidence=missing,
        assumptions=assumptions,
        phase=input_data.phase,
        week_number=input_data.week_number,
        planned_workout_id=input_data.planned_workout_id,
    )
```

```python
from __future__ import annotations

from pathlib import Path

import yaml
from pydantic import BaseModel

from running_coach.models import SourceType


class ScienceRef(BaseModel):
    science_ref_id: str
    title: str
    authors: str
    year: int
    source_type: SourceType
    journal_or_publisher: str
    doi_or_url: str
    population: str
    finding: str
    practical_application: str
    limits: str
    tags: list[str]
    approved: bool
    approved_date: str | None
    notes: str | None = None


def load_science_refs(path: Path) -> dict[str, ScienceRef]:
    data = yaml.safe_load(path.read_text(encoding="utf-8"))
    refs = [ScienceRef.model_validate(item) for item in data["refs"]]
    return {ref.science_ref_id: ref for ref in refs}


def validate_science_tags(
    refs: dict[str, ScienceRef],
    science_ref_ids: list[str],
    required_tags: list[str],
) -> None:
    required = set(required_tags)
    for ref_id in science_ref_ids:
        ref = refs[ref_id]
        if not ref.approved:
            raise ValueError(f"science ref is not approved: {ref_id}")
        if not required.intersection(ref.tags):
            raise ValueError(f"science ref {ref_id} does not match required tags {required_tags}")
```

```python
from __future__ import annotations

import csv
from pathlib import Path

from running_coach.garmin import parse_garmin_csv_text
from running_coach.reports import write_text_report


def write_csv(path: Path, rows: list[dict[str, object]]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    if not rows:
        path.write_text("", encoding="utf-8")
        return
    with path.open("w", newline="", encoding="utf-8") as handle:
        writer = csv.DictWriter(handle, fieldnames=list(rows[0].keys()))
        writer.writeheader()
        writer.writerows(rows)


def run_pipeline(garmin_csv: Path, repo_root: Path, after_workout: bool) -> None:
    activities = parse_garmin_csv_text(garmin_csv.read_text(encoding="utf-8-sig"), garmin_csv.name)
    processed = repo_root / "data/processed"
    write_csv(processed / "activities.csv", activities)
    workouts = [
        {
            "local_date": row["local_date"],
            "activity_id": row["activity_id"],
            "participants": '["matheus"]',
            "shared_run": "false",
            "bruna_present": "false",
            "matheus_role": "solo",
            "confidence": "low",
            "evidence_level": "low",
            "missing_evidence": '["checkin"]',
        }
        for row in activities
    ]
    write_csv(processed / "workouts.csv", workouts)
    write_csv(processed / "decisions.csv", [{"date": activities[0]["local_date"] if activities else "", "event": "pipeline", "decision": "initialized"}])
    write_csv(processed / "science_refs.csv", [])
    write_text_report(repo_root / "docs/state.md", "Current State", ["Pipeline initialized from Garmin data."])
    write_text_report(repo_root / "docs/decisions.md", "Decisions", ["- Initial pipeline decision log generated."])
    write_text_report(repo_root / "reports/latest-summary.md", "Latest Summary", ["Recommendation confidence depends on check-in completeness."])
```

```python
from __future__ import annotations

from pathlib import Path

from openpyxl import Workbook
from openpyxl.chart import LineChart, Reference
from openpyxl.styles import Font, PatternFill


SHEETS = [
    "Dashboard",
    "Estado Atual",
    "Próximos Treinos",
    "Treinos Garmin",
    "Semanas",
    "Provas e Marcos",
    "Decisões",
    "Roadmap",
    "Ciência & Prompt",
    "Dados Gráficos",
]


def style_header(ws) -> None:
    fill = PatternFill("solid", fgColor="0B1F33")
    font = Font(color="FFFFFF", bold=True)
    for cell in ws[1]:
        cell.fill = fill
        cell.font = font


def build_dashboard(output: Path) -> None:
    wb = Workbook()
    wb.remove(wb.active)
    for name in SHEETS:
        wb.create_sheet(name)

    dashboard = wb["Dashboard"]
    dashboard["A1"] = "Projeto Meia Forte Janeiro 2027 — Matheus & Bruna"
    dashboard["A1"].font = Font(size=18, bold=True, color="0B1F33")
    cards = [
        ("A3", "Evolução", "ON TRACK"),
        ("C3", "Fadiga", "ATENÇÃO"),
        ("E3", "Risco", "CONTROLADO"),
        ("G3", "Próximo treino", "10K controlado"),
        ("I3", "Meia forte", "Hipótese"),
    ]
    for cell, title, value in cards:
        dashboard[cell] = title
        dashboard[cell].fill = PatternFill("solid", fgColor="E8F1F2")
        dashboard[cell].font = Font(bold=True, color="0B1F33")
        dashboard[cell.replace("3", "4")] = value

    data = wb["Dados Gráficos"]
    data.append(["Data", "Pace"])
    data.append(["2026-05-03", 6.8])
    data.append(["2026-05-10", 6.6])
    data.append(["2026-05-28", 6.1])
    style_header(data)

    chart = LineChart()
    chart.title = "Tendência pace forte"
    chart.y_axis.title = "min/km"
    chart.x_axis.title = "Data"
    chart.add_data(Reference(data, min_col=2, min_row=1, max_row=4), titles_from_data=True)
    chart.set_categories(Reference(data, min_col=1, min_row=2, max_row=4))
    dashboard.add_chart(chart, "A8")

    for name in SHEETS:
        ws = wb[name]
        if ws.max_row == 1 and ws["A1"].value is None:
            ws.append([name])
        style_header(ws)
        ws.freeze_panes = "A2"
        ws.auto_filter.ref = ws.dimensions

    output.parent.mkdir(parents=True, exist_ok=True)
    wb.save(output)
```
